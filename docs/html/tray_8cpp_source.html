41/**
42 * @class YggdrasilTray

45 * This class handles GUI interactions, service toggling, and updates

48class YggdrasilTray : public QObject {

51public:
52 /**
53 * @brief Constructs the YggdrasilTray object.
54 * @param parent Pointer to the parent QObject, if any.
55 */
56 explicit YggdrasilTray(QObject *parent = nullptr)
57 : QObject(parent),
58 serviceManager("yggdrasil"),
59 socketManager(YGG_SOCKET_PATH) {
60 trayIcon = new QSystemTrayIcon(this);
61 trayIcon->setIcon(QIcon::fromTheme(ICON_NOT_RUNNING));
62 trayIcon->setToolTip(TOOLTIP);
63
64 trayMenu = new QMenu();
65
66 // Status menu item
67 statusAction = new QAction("Status: Unknown", trayMenu);
68 statusAction->setDisabled(true);
69 trayMenu->addAction(statusAction);
70
71 // IP address menu item
72 ipAction = new QAction("IP: Unknown", trayMenu);
73 ipAction->setDisabled(true);
74 trayMenu->addAction(ipAction);
75
76 trayMenu->addSeparator();
77
78 // Toggle Yggdrasil service action
79 toggleAction = new QAction("Start/Stop Yggdrasil", trayMenu);
80 connect(toggleAction, &QAction::triggered, this, &YggdrasilTray::toggleYggdrasilService);
81 trayMenu->addAction(toggleAction);
82
83 // Copy IP action
84 copyIPAction = new QAction("Copy IP", trayMenu);
85 connect(copyIPAction, &QAction::triggered, this, &YggdrasilTray::copyIP);
86 trayMenu->addAction(copyIPAction);
87
88 trayMenu->addSeparator();
89
90 // Quit action
91 QAction *quitAction = new QAction("Quit", trayMenu);
92 connect(quitAction, &QAction::triggered, qApp, &QApplication::quit);
93 trayMenu->addAction(quitAction);

95 trayIcon->setContextMenu(trayMenu);
96 trayIcon->show();
97
98 // Connect activated signal
99 connect(trayIcon, &QSystemTrayIcon::activated, this, &YggdrasilTray::onTrayIconActivated);
100
101 // Periodic update
102 QTimer *timer = new QTimer(this);
103 connect(timer, &QTimer::timeout, this, &YggdrasilTray::updateTrayIcon);
104 timer->start(5000);

106 updateTrayIcon();

109private slots:
110 /**
111 * @brief Toggles the Yggdrasil service between running and stopped states.
112 *
113 * This function checks the current status of the Yggdrasil service
114 * and either starts or stops it. It updates the tray icon and displays
115 * a message indicating the result of the operation.
116 */
117 void toggleYggdrasilService() {
118 bool success;
119 QString action;
120
121 if (serviceManager.isServiceRunning()) {
122 success = serviceManager.stopService();
123 action = "stopped";
124 } else {
125 success = serviceManager.startService();
126 action = "started";
127 }
128
129 if (success) {
130 QMessageBox::information(nullptr, "Service Toggle", "Yggdrasil service successfully " + action + ".");
131 } else {
132 QMessageBox::critical(nullptr, "Service Toggle", "Failed to toggle Yggdrasil service.");
133 }
134
135 updateTrayIcon();
136 }
137
138 /**
139 * @brief Copies the Yggdrasil IP address to the clipboard.
140 *
141 * Retrieves the Yggdrasil IP address from the socket manager and
142 * copies it to the system clipboard. Displays a message indicating
143 * the result of the operation.
144 */
145 void copyIP() {
146 QString ip = socketManager.getYggdrasilIP();
147 if (!ip.isEmpty()) {
148 QApplication::clipboard()->setText(ip);
149 QMessageBox::information(nullptr, "Copy IP", "IP copied to clipboard: " + ip);
150 } else {
151 QMessageBox::warning(nullptr, "Copy IP", "Failed to retrieve IP.");
152 }
153 }
154
155 /**
156 * @brief Updates the tray icon and menu items.
157 *
158 * Retrieves the current Yggdrasil service status and IP address,
159 * then updates the tray icon and menu items to reflect this information.
160 */
161 void updateTrayIcon() {
162 QString status = serviceManager.isServiceRunning() ? "Running" : "Not Running";
163 QString ip = socketManager.getYggdrasilIP();
164
165 statusAction->setText("Status: " + status);
166 ipAction->setText("IP: " + ip);
167
168 trayIcon->setIcon(QIcon::fromTheme(status == "Running" ? ICON_RUNNING : ICON_NOT_RUNNING));
169 }
170
171 /**
172 * @brief Handles tray icon activation events.
173 * @param reason The reason for the activation event.
174 *
175 * Displays the tray menu when the tray icon is activated
176 * via a left or right click.
177 */
178 void onTrayIconActivated(QSystemTrayIcon::ActivationReason reason) {
179 if (reason == QSystemTrayIcon::Trigger || reason == QSystemTrayIcon::Context) {
180 trayMenu->popup(QCursor::pos());
181 }
182 }
183
184private:
185 QSystemTrayIcon *trayIcon; ///< The system tray icon.
186 QMenu *trayMenu; ///< The menu displayed in the system tray.
187 QAction *statusAction; ///< Action displaying the service status.
188 QAction *ipAction; ///< Action displaying the Yggdrasil IP address.
189 QAction *toggleAction; ///< Action to toggle the Yggdrasil service.
190 QAction *copyIPAction; ///< Action to copy the Yggdrasil IP address.
191 ServiceManager serviceManager; ///< Manager for service-related tasks.
192 SocketManager socketManager; ///< Manager for socket-related tasks.
193};

205int main(int argc, char *argv[]) {
206 QApplication app(argc, argv);
207
