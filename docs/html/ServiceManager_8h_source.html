19 * This class provides methods to check the status of a service, start

22class ServiceManager {
23public:
24 /**
25 * @brief Constructs a ServiceManager for the specified service.
26 * @param serviceName The name of the service to manage.
27 */
28 explicit ServiceManager(const QString &serviceName) : serviceName(serviceName) {}

30 /**
31 * @brief Checks if the service is currently running.
32 * @return True if the service is running, false otherwise.
33 *
34 * Uses the `systemctl is-active` command to determine the status
35 * of the specified service.
36 */
37 bool isServiceRunning() const {
38 QProcess process;
39 QStringList arguments = {"is-active", serviceName};
40 process.start("systemctl", arguments);
41 process.waitForFinished();
42
43 QString output = process.readAllStandardOutput().trimmed();
44 return (output == "active");
45 }
46
47 /**
48 * @brief Starts the specified service.
49 * @return True if the service was started successfully, false otherwise.
50 *
51 * Executes the `systemctl start` command using pkexec.
52 */
53 bool startService() const {
54 return executeCommand("start");
55 }
56
57 /**
58 * @brief Stops the specified service.
59 * @return True if the service was stopped successfully, false otherwise.
60 *
61 * Executes the `systemctl stop` command using pkexec.
62 */
63 bool stopService() const {
64 return executeCommand("stop");
65 }
66
67private:
68 QString serviceName; ///< The name of the service to manage.
69
70 /**
71 * @brief Executes a systemctl command for the specified action.
72 * @param action The action to perform (e.g., "start" or "stop").
73 * @return True if the command was successful, false otherwise.
74 *
75 * This method uses pkexec to run the systemctl command with
76 * elevated privileges.
77 */
78 bool executeCommand(const QString &action) const {
79 QProcess process;
80 QStringList arguments = {"systemctl", action, serviceName};
81 process.start("pkexec", arguments);
82 process.waitForFinished();
83
84 if (process.exitCode() == 0) {
85 qDebug() << action << "command executed successfully for" << serviceName;
86 return true;
87 } else {
88 qDebug() << action << "command failed for" << serviceName << ":" << process.readAllStandardError();
89 return false;
90 }
91 }
